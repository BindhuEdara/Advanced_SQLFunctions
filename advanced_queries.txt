------------Step 2: Aggregate Queries------------------

-- Count the total number of students
SELECT COUNT(*) AS total_students
FROM students


-- Find the average age of students
SELECT SUM(age)/COUNT(*) FROM students


-- Find the minimum and maximum age

SELECT MIN(AGE),MAX(AGE) FROM students

-- Count how many students are enrolled in each course
SELECT course,COUNT(*) FROM STUDENTS GROUP BY course


-- Count students whose age is greater than 20
SELECT COUNT(*) FROM students WHERE AGE>20


------------ STEP:3 GROUPING AND FILTERING --------------

-- Display course-wise student count
SELECT course,COUNT(*) FROM students GROUP BY COURSE

-- Display only courses where student count is greater than 1
SELECT COURSE  FROM students group by course having COUNT(COURSE)>1

-- Display courses sorted by highest student count first
SELECT course FROM STUDENTS GROUP BY COURSE order by COUNT(students) desc


------------ Step 4: Ordering & Limiting---------------

-- Fetch the latest 3 students based on created_at
SELECT * FROM students ORDER BY created_at DESC LIMIT 3

-- Fetch the top 5 oldest students
SELECT * FROM students ORDER BY created_at ASC LIMIT 5

-- Fetch students enrolled in a specific course, sorted by age (descending)
SELECT name, COURSE FROM students WHERE COURSE='Computers' order by age desc
